function arrayMin(arr) {
 
            min = arr[0]
            for(i=0; i < arr.length; i++){
              if(arr[i] < min){
                min=arr[i]
              }
            }
            return min
           }

11.Birthday Game

function Birthday_Game(arr,D ,M) {
 
  count=0
  for(let i=0; i < arr.length-M+1; i++){
    let sum=0
    for(let j=i; j < M+i; j++){
      sum+=arr[j]
    }
    if (sum === D){
      count++
    }
  }
  return count
  }
